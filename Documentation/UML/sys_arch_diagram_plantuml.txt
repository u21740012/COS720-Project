    component "Prediction Controller" as Controller

    component "Prediction Service" as Predictor
    component "Explanation Service" as Explain

    component "Data Preprocessing" as Preprocess
    component "Classification Model" as Model
}

    database "Input Data" as Uploads
    database "Prediction Logs" as Logs
    database "Training Dataset" as TrainData
    database "Testing Dataset" as TestData
}

' Force stacked layers
PL -[hidden]down-> AL

Predictor -[hidden]right-> Explain

' Keep data stores on same horizontal level
Uploads -[hidden]right-> Logs
Logs -[hidden]right-> TrainData
TrainData -[hidden]right-> TestData

' Main flow
User --> UI
UI --> Controller
Controller --> Predictor
Predictor --> Preprocess
Preprocess --> Model
Model --> Predictor
Predictor --> Controller
Controller --> UI

' Supporting flows
Controller --> Explain
Controller --> Logger
Logger --> Logs

Uploads --> UI
TrainData --> Model
